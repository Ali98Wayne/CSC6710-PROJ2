CREATE DATABASE proj_2;
USE proj_2;

CREATE TABLE Users(
   user_id INT AUTO_INCREMENT PRIMARY KEY,
   username VARCHAR(50),
   password VARCHAR(60),
   first_name VARCHAR(50),
   last_name VARCHAR(50),
   address_street VARCHAR(50),
   address_city VARCHAR(50),
   address_state VARCHAR(2),
   address_zip VARCHAR(5),
   phone VARCHAR(12),
   email VARCHAR(50),
   credit_card_num VARCHAR(50),
   credit_card_month VARCHAR(20),
   credit_card_year VARCHAR(4),
   credit_card_cvv VARCHAR(4),
   signup_date DATE,
   last_login DATETIME,
   order_generated TINYINT(1) DEFAULT 0,
   bill_generated TINYINT(1) DEFAULT 0
)

CREATE TABLE Request_Cleaning(
   request_id INT AUTO_INCREMENT PRIMARY KEY,
   client_id INT NOT NULL,
   service_address_street VARCHAR(50),
   service_address_city VARCHAR(50),
   service_address_state VARCHAR(2),
   service_address_zip VARCHAR(5),
   cleaning_type ENUM('Basic', 'Deep Cleaning', 'Move-Out') NOT NULL,
   rooms INT(20),
   preferred_date DATETIME,
   proposed_budget FLOAT,
   notes TEXT NULL,
   photo_urls JSON NULL;
   request_date DATE DEFAULT CURRENT_DATE,
   quote_accept_date DATE,
   bill_status ENUM('Paid', 'Unpaid'),
   bill_due_date DATE,
   FOREIGN KEY (client_id) REFERENCES Users(user_id)
)
